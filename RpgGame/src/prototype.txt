public void removeXp1(int amount) {
		int tmp = this.finalXp;
		if (this.finalXp == 0 && this.finalLvl == 0) {
			System.err.println("Lvl and xp is already at 0");
		}
		this.finalXp -= amount;
		while (this.finalXp < 0) {
			if (this.finalXp == 0 && this.finalLvl == 0) {
				System.out.println("Lvl and xp are now null");
				break;
			}
			this.finalLvl -= 1;
			this.finalXp = baseAmountLvlUp - (amount - tmp);
		}
		updateStat();
	}
	
	public void removeXp2(int amount) {
		if (this.finalXp == 0 && this.finalLvl == 0) {
			System.err.println("Lvl and xp is already at 0");
			return;
		}
		if (amount == this.finalXp) {
			this.finalXp = 0;
		}
		else if (amount > this.finalXp) {
			if (amount < baseAmountLvlUp) {
				if (this.finalLvl == 0) {
					this.finalXp = 0;
				}
				this.finalLvl -= 1;
				this.finalXp = baseAmountLvlUp + (this.finalXp - amount);
			}
			while (amount >= baseAmountLvlUp) {
				this.finalXp -= baseAmountLvlUp;
				amount -= baseAmountLvlUp;
				if (this.finalLvl == 0) {
					this.finalXp = 0;
					break;
				}
				this.finalLvl -= 1;
				this.finalXp += baseAmountLvlUp;
			}
		}
		else this.finalXp -= amount;
		updateStat();
	}
	
	public void removeXp3(int amount) {
		if (this.finalXp == 0 && this.finalLvl == 0) {
			System.err.println("Lvl and xp is already at 0");
			return;
		}
		else if (this.finalXp >= amount) {
			this.finalXp -= amount;
		}
		else 
			while (amount >= baseAmountLvlUp) {
				this.finalXp -= amount;
				int tmp = this.finalXp;
				this.finalXp = baseAmountLvlUp - tmp;
				this.finalLvl -= 1;
			}
		updateStat();
	}
